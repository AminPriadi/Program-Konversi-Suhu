    #include<iostream>
    using namespace std;

    int main ()
    {
        int celcius,farenheit,reamur;
        cout<<"masukan suhu celcius =";
        cin>>celcius;
        cout<<"\n\t";
         farenheit=9/5*celcius+32;
         cout<<"suhu farenheitnya adalah =";
         cout<<farenheit<<endl;

         reamur=0.8*celcius;
         cout<<"suhu reamurnya adalah =";
         cout<<reamur;

         return 0;
    }
